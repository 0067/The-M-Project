 11 m_require('core/foundation/request.js');
 12 
 13 /**
 14  * Logging levels as constants.
 15  */
 16 M.DEBUG = 0;
 17 M.ERROR = 1;
 18 M.WARN = 2;
 19 M.INFO = 3;
 20 
 21 /**
 22  * @class
 23  *
 24  * Object for logging.
 25  *
 26  */
 27 M.Logger = M.Object.extend(
 28 /** @scope M.Logger.prototype */ {
 29 
 30     /**
 31      * The type of this object.
 32      *
 33      * @property {String}
 34      */
 35     type: 'M.Logger',
 36 
 37     /**
 38      * Possible values for the logging level:
 39      *
 40      * - debug:   M.DEBUG
 41      * - error:   M.ERROR
 42      * - warning: M.WARNING
 43      *
 44      * @param {String} msg The logging message.
 45      * @param {Number} level The logging level.
 46      */
 47     log: function(msg, level) {
 48         level = level || M.DEBUG;
 49 
 50         /* Prevent a console.log from blowing things up if we are on a browser that doesn't support this. */
 51         if (typeof console === 'undefined') {
 52             window.console = {} ;
 53             console.log = console.info = console.warn = console.error = function(){};
 54         }
 55         
 56         switch (level) {
 57             case M.DEBUG:
 58                 this.debug(msg);
 59                 break;
 60             case M.ERROR:
 61                 this.error(msg);
 62                 break;
 63             case M.WARN:
 64                 this.warn(msg);
 65                 break;
 66             case M.INFO:
 67                 this.info(msg);
 68                 break;
 69             default:
 70                 this.debug(msg);
 71                 break;
 72         }
 73     },
 74 
 75     /**
 76      * @private
 77      */
 78     debug: function(msg) {
 79         console.debug(msg);
 80     },
 81 
 82     /**
 83      * @private
 84      */
 85     error: function(msg) {
 86         console.error(msg);
 87     },
 88 
 89     /**
 90      * @private
 91      */
 92     warn: function(msg) {
 93         console.warn(msg);
 94     },
 95 
 96     /**
 97      * @private
 98      */
 99     info: function(msg) {
100         console.info(msg);
101     }
102 
103 });
