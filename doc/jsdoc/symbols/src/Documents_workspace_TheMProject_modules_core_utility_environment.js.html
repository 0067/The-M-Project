 11 m_require('core/foundation/object.js');
 12 
 13 M.OFFLINE = 'offline';
 14 M.ONLINE = 'online';
 15 
 16 
 17 /**
 18  * @class
 19  *
 20  * Encapsulates methods to retrieve information about the environment,
 21  * like browser used, platform, user agent (based on navigator object) or
 22  * if the device is on- or offline (determined via ajax request).
 23  */
 24 M.Environment = M.Object.extend(
 25 /** @scope M.Environment.prototype */ {
 26 
 27     /**
 28      * Checks the connection status by sending an ajax request
 29      * and waiting for the response to decide whether online or offline.
 30      *
 31      * The callback is called when the request returns successful or times out. The parameter to callback is a
 32      * string saying either offline or online.
 33      *
 34      * @param {function} callback The function to be called when request returns.
 35      * @param {String} url Optional. The request url. When not given, request is made to google.com
 36      * @param {Number} timeout Optional. Time in milliseconds until request is considered to be timed out. Defaults to 5 seconds.
 37      */
 38     getConnectionStatus: function(callback, url, timeout){
 39         M.Request.init({
 40                 url: url ? url : '/google',
 41                 isJSON: NO,
 42                 timeout: timeout ? timeout : 5000,
 43                 onSuccess: function(data){
 44                     callback(M.ONLINE);
 45                 },
 46                 onError: function(data){
 47                     callback(M.OFFLINE);
 48                 }
 49             }).send();
 50     },
 51 
 52     /**
 53      * Returns the userAgent as received from navigator object.
 54      * E.g. "Mozilla/5.0 (Macintosh; U; Intel Mac OS X 10_6_5; en-US) AppleWebKit/534.7 (KHTML, like Gecko) Chrome/7.0.517.44 Safari/534.7"
 55      */
 56     getUserAgent: function() {
 57         return navigator.userAgent;
 58     },
 59 
 60     /**
 61      * Returns the platform as received from navigator object.
 62      * E.g. "MacIntel"
 63      */
 64     getPlatform: function() {
 65         return navigator.platform;
 66     },
 67 
 68     /**
 69      * Returns the browser version as received from navigator object.
 70      * E.g. "5.0 (Macintosh; U; Intel Mac OS X 10_6_5; en-US) AppleWebKit/534.7 (KHTML, like Gecko) Chrome/7.0.517.44 Safari/534.7"
 71      */
 72     getBrowserName: function() {
 73         return navigator.appName;
 74     },
 75 
 76     /**
 77      * Returns the currently available width and height of the browser window
 78      * as an array:
 79      *
 80      * 0 -> width
 81      * 1 -> height
 82      */
 83     getSize: function() {
 84         var viewportWidth;
 85         var viewportHeight;
 86 
 87         if(typeof window.innerWidth != 'undefined') {
 88             viewportWidth = window.innerWidth,
 89             viewportHeight = window.innerHeight
 90         } else if(typeof document.documentElement != 'undefined' && typeof document.documentElement.clientWidth != 'undefined' && document.documentElement.clientWidth != 0) {
 91             viewportWidth = document.documentElement.clientWidth,
 92             viewportHeight = document.documentElement.clientHeight
 93         } else {
 94             viewportWidth = document.getElementsByTagName('body')[0].clientWidth,
 95             viewportHeight = document.getElementsByTagName('body')[0].clientHeight
 96         }
 97 
 98         return [viewportWidth, viewportHeight];
 99     },
100 
101     /**
102      * Returns the currently available width of the browser window.
103      */
104     getWidth: function() {
105         return this.getSize()[0];
106     },
107 
108     /**
109      * Returns the currently available height of the browser window.
110      */
111     getHeight: function() {
112         return this.getSize()[1];
113     }
114 
115 
116 });
