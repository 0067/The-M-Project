 67     /**
 68      * This method returns a date for a given date string. It is based on JS Date's parse
 69      * method.
 70      *
 71      * The following formats are accepted (time and timezone are optional):
 72      * - 11/12/2010
 73      * - 11/12/2010 15:28:15
 74      * - 11/12/2010 13:28:15 GMT
 75      * - 11/12/2010 15:28:15 GMT+0200
 76      * - 12 November 2010
 77      * - 12 November 2010 15:28:15
 78      * - 12 November 2010 13:28:15 GMT
 79      * - 12 November 2010 15:28:15 GMT+0200
 80      * - 12 Nov 2010
 81      * - 12 Nov 2010 15:28:15
 82      * - 12 Nov 2010 13:28:15 GMT
 83      * - 12 Nov 2010 15:28:15 GMT+0200
 84      *
 85      * If a wrong formatted date string is given, the method will return null. Otherwise a
 86      * JS Date object will be returned.
 87      */
 88     create: function(dateString) {
 89         var milliseconds = typeof(dateString) === 'number' ? dateString : null;
 90 
 91         if(!milliseconds) {
 92             var regexResult = /(\d{1,2})\.(\d{1,2})\.(\d{2,4})/.exec(dateString);
 93             if(regexResult && regexResult[1] && regexResult[2] && regexResult[3]) {
 94                 var date = $.trim(dateString).split(' ');
 95                 dateString = regexResult[2] + '/' + regexResult[1] + '/' + regexResult[3] + (date[1] ? ' ' + date[1] : '');
 96             }
 97             milliseconds = Date.parse(dateString);
 98         }
 99 
100         if(dateString && !milliseconds) {
101             M.Logger.log('Invalid dateString \'' + dateString + '\'.', M.WARN);
102             return null;
103         } else if(!dateString) {
104             return this.now();
105         }
106 
107         return this.extend({
108             date: new Date(milliseconds)
109         });
110     },

112     /**
113      * This method formats a given date object according to the passed 'format' property and
114      * returns it as a String.
115      *
116      * The following list defines the special characters that can be used in the 'format' property
117      * to format the given date:
118      *
119      * d        Day of the month as digits; no leading zero for single-digit days.
120      * dd 	    Day of the month as digits; leading zero for single-digit days.
121      * ddd 	    Day of the week as a three-letter abbreviation.
122      * dddd 	Day of the week as its full name.
123      * m 	    Month as digits; no leading zero for single-digit months.
124      * mm 	    Month as digits; leading zero for single-digit months.
125      * mmm 	    Month as a three-letter abbreviation.
126      * mmmm 	Month as its full name.
127      * yy 	    Year as last two digits; leading zero for years less than 10.
128      * yyyy 	Year represented by four digits.
129      * h 	    Hours; no leading zero for single-digit hours (12-hour clock).
130      * hh 	    Hours; leading zero for single-digit hours (12-hour clock).
131      * H 	    Hours; no leading zero for single-digit hours (24-hour clock).
132      * HH 	    Hours; leading zero for single-digit hours (24-hour clock).
133      * M 	    Minutes; no leading zero for single-digit minutes.
134      * MM 	    Minutes; leading zero for single-digit minutes.
135      * s 	    Seconds; no leading zero for single-digit seconds.
136      * ss 	    Seconds; leading zero for single-digit seconds.
137      * l or L 	Milliseconds. l gives 3 digits. L gives 2 digits.
138      * t 	    Lowercase, single-character time marker string: a or p.
139      * tt   	Lowercase, two-character time marker string: am or pm.
140      * T 	    Uppercase, single-character time marker string: A or P.
141      * TT 	    Uppercase, two-character time marker string: AM or PM.
142      * Z 	    US timezone abbreviation, e.g. EST or MDT. With non-US timezones or in the Opera browser, the GMT/UTC offset is returned, e.g. GMT-0500
143      * o 	    GMT/UTC timezone offset, e.g. -0500 or +0230.
144      * S 	    The date's ordinal suffix (st, nd, rd, or th). Works well with d.
145      *
146      * @param format {String} The format.
147      * @param utc {Boolean} Determines whether to convert to UTC time or not. Default: NO.
148      */
149     format: function(format, utc) {
150         if(isNaN(this.date)) {
151             M.Logger.log('Invalid date!', M.WARN);   
152         }
153 
154         var	token = /d{1,4}|m{1,4}|yy(?:yy)?|([HhMsTt])\1?|[LloSZ]|"[^"]*"|'[^']*'/g;
155         var	timezone = /\b(?:[PMCEA][SDP]T|(?:Pacific|Mountain|Central|Eastern|Atlantic) (?:Standard|Daylight|Prevailing) Time|(?:GMT|UTC)(?:[-+]\d{4})?)\b/g;
156         var	timezoneClip = /[^-+\dA-Z]/g;
157         var	pad = function (val, len) {
158             val = String(val);
159             len = len || 2;
160             while (val.length < len) val = "0" + val;
161             return val;
162         };
163 
164 		if(arguments.length == 1 && Object.prototype.toString.call(this.date) == "[object String]" && !/\d/.test(this.date)) {
165 			format = this.date;
166 			date = undefined;
167 		}
168 
169 		var	_ = utc ? "getUTC" : "get";
170         var	d = this.date[_ + "Date"]();
171         var	D = this.date[_ + "Day"]();
172         var	m = this.date[_ + "Month"]();
173         var	y = this.date[_ + "FullYear"]();
174         var	H = this.date[_ + "Hours"]();
175         var	Min = this.date[_ + "Minutes"]();
176         var	s = this.date[_ + "Seconds"]();
177         var	L = this.date[_ + "Milliseconds"]();
178         var	o = utc ? 0 : this.date.getTimezoneOffset();
179         var	flags = {
180             d:    d,
181             dd:   pad(d),
182             ddd:  M.DAY_NAMES[D],
183             dddd: M.DAY_NAMES[D + 7],
184             m:    m + 1,
185             mm:   pad(m + 1),
186             mmm:  M.MONTH_NAMES[m],
187             mmmm: M.MONTH_NAMES[m + 12],
188             yy:   String(y).slice(2),
189             yyyy: y,
190             h:    H % 12 || 12,
191             hh:   pad(H % 12 || 12),
192             H:    H,
193             HH:   pad(H),
194             M:    Min,
195             MM:   pad(Min),
196             s:    s,
197             ss:   pad(s),
198             l:    pad(L, 3),
199             L:    pad(L > 99 ? Math.round(L / 10) : L),
200             t:    H < 12 ? "a"  : "p",
201             tt:   H < 12 ? "am" : "pm",
202             T:    H < 12 ? "A"  : "P",
203             TT:   H < 12 ? "AM" : "PM",
204             Z:    utc ? "UTC" : (String(this.date).match(timezone) || [""]).pop().replace(timezoneClip, ""),
205             o:    (o > 0 ? "-" : "+") + pad(Math.floor(Math.abs(o) / 60) * 100 + Math.abs(o) % 60, 4),
206             S:    ["th", "st", "nd", "rd"][d % 10 > 3 ? 0 : (d % 100 - d % 10 != 10) * d % 10]
207         };
208         
209 		return format.replace(token, function ($0) {
210 			return $0 in flags ? flags[$0] : $0.slice(1, $0.length - 1);
211 		});
212     },

325     /**
326      * This method returns a date in the future or past, based on 'milliseconds' and a given date. Basically
327      * it adds or subtracts x milliseconds, but also checks for clock changes and automatically includes
328      * these into the calculation of the future or past date.
329      *
330      * @param {Number} milliseconds The number of milliseconds to be added to or subtracted from the current date.
331      */
332     millisecondsFromDate: function(milliseconds) {
333         if(!this.date) {
334             M.Logger.log('no date specified!', M.ERROR);
335         }
336 
337         var outputDate = new Date(Date.parse(this.date) + milliseconds);
338         return this.extend({
339             date: new Date(Date.parse(outputDate) + (outputDate.getTimezoneOffset() - this.date.getTimezoneOffset()) * (60 * 1000))
340         });
341     },

343     /**
344      * This method returns the time between two dates, based on the given returnType.
345      *
346      * Possible returnTypes are:
347      * - M.MILLISECONDS: milliseconds
348      * - M.SECONDS: seconds
349      * - M.MINUTES: minutes
350      * - M.HOURS: hours
351      * - M.DAYS: days
352      *
353      * @param {Object} date The date.
354      * @param {String} returnType The return type for the call.
355      */
356     timeBetween: function(date, returnType) {
357         var firstDateInMilliseconds = this.date ? this.date.valueOf() : null;
358         var secondDateInMilliseconds = date.date ? date.date.valueOf() : null;
359         
360         if(firstDateInMilliseconds && secondDateInMilliseconds) {
361             switch (returnType) {
362                 case M.DAYS:
363                     return (secondDateInMilliseconds - firstDateInMilliseconds) / (24 * 60 * 60 * 1000);
364                     break;
365                 case M.HOURS:
366                     return (secondDateInMilliseconds - firstDateInMilliseconds) / (60 * 60 * 1000);
367                     break;
368                 case M.MINUTES:
369                     return (secondDateInMilliseconds - firstDateInMilliseconds) / (60 * 1000);
370                     break;
371                 case M.SECONDS:
372                     return (secondDateInMilliseconds - firstDateInMilliseconds) / 1000;
373                     break;
374                 case M.MILLISECONDS:
375                 default:
376                     return (secondDateInMilliseconds - firstDateInMilliseconds);
377                     break;
378             }
379         } else if(firstDateInMilliseconds) {
380             M.Logger.log('invalid date passed.', M.ERROR);
381         } else {
382             M.Logger.log('invalid date.', M.ERROR);
383         }
384     },
