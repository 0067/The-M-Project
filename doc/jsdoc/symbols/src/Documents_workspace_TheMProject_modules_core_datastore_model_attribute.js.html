 11 m_require('core/utility/logger.js');

 14 /**
 15  * @class
 16  *
 17  * Defines an attribute for a model
 18  */
 19 M.ModelAttribute = M.Object.extend(
 20 /** @scope M.ModelAttribute.prototype */ {
 21 
 22     type: 'M.ModelAttribute',
 23 
 24     /**
 25      * The data type for the model record property.
 26      * Extremely important e.g. to map model to relational database table.
 27      *
 28      * @type String
 29      */
 30     dataType: null,
 31 
 32     /**
 33      * Indicates whether this property is required to be set before persisting.
 34      * If YES, then automatically @link M.PresenceValidator is added to the property, to check the presence.
 35      * 
 36      * @type Boolean
 37      */
 38     isRequired: NO,
 39 
 40     /**
 41      * Array containing all validators for this model record property.
 42      * E.g. [@link M.PresenceValidator, @link M.NumberValidator]
 43      * @type Object
 44      */
 45     validators: null,
 46 
 47     /**
 48      * Iterates over validators array and calls validate on each validator with the param object passed to the validator.
 49      * @param {Object} obj The parameter object containing the model id, the record as M.ModelAttribute object and the value of the property.
 50      * @return {Boolean} Indicates wheter the property is valid (YES|true) or invalid (NO|false). 
 51      */
 52     validate: function(obj) {
 53         var isValid = YES;
 54         for (var i in this.validators) {
 55             if(!this.validators[i].validate(obj)) {
 56                isValid = NO; 
 57             }
 58         }
 59         return isValid;
 60     }
 61 });
 62 
 63 //
 64 // CLASS METHODS
 65 //
 66 
 67 /**
 68  * Returns a model attribute.
 69  *
 70  * @param dataType The data type of the attribute: e.g. String 
 71  * @param opts options for the attribute, such as defaultValue, isRequired flag, etc. ...
 72  * @return {Object} Model attribute
 73  */
 74 M.ModelAttribute.attr = function(dataType, opts) {
 75     if (!opts) {
 76         opts = {};
 77     }
 78     if (!opts.dataType) {
 79         opts.dataType = dataType || 'String';
 80     }
 81 
 82     /* if validators array is not set and attribute is required, define validators as an empty array, (this is for adding M.PresenceValidator automatically */
 83     if (!opts.validators && opts.isRequired) {
 84         opts.validators = [];
 85     }
 86 
 87     /* if model attribute is required, presence validator is automatically inserted */
 88     if (opts.isRequired) {
 89         /* check if custom presence validator has been added to validators array, if not add the presence validator*/
 90         if( _.select(opts.validators, function(v){return v.type === 'M.PresenceValidator'}).length === 0) {
 91             opts.validators.push(M.PresenceValidator);
 92         }
 93     }
 94     return this.extend(opts);
 95 };
