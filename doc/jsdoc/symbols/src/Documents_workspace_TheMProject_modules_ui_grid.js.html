 27 /**
 28  * @class
 29  *
 30  * The root object for GridViews.
 31  *
 32  */
 33 M.GridView = M.View.extend(
 34 /** @scope M.GridView.prototype */ {
 35 
 36     type: 'M.GridView',
 37 
 38     layout: null,
 39 
 40     /**
 41      * Renders a GridView based on the specified layout.
 42      */
 43     render: function() {
 44         this.html += '<div id="' + this.id + '" ' + this.style() + '>';
 45 
 46         this.renderChildViews();
 47         
 48         this.html += '</div>';
 49 
 50         return this.html;
 51     },
 52 
 53     /**
 54      * Triggers render() on all children and includes some special GridView logic
 55      * concerning the rendering of these child views.
 56      */
 57     renderChildViews: function() {
 58         if(this.childViews) {
 59             if(this.layout) {
 60                 var arr = this.childViews.split(' ');
 61                 for(var i in this.layout.columns) {
 62                     if(this[arr[i]]) {
 63                         this.html += '<div class="' + this.layout.columns[i] + '">';
 64 
 65                         this.html += this[arr[i]].render();
 66 
 67                         this.html += '</div>';
 68                     }
 69                 }
 70             } else {
 71                 M.Logger.log('No layout specified for GridView', M.ERROR);
 72             }
 73         }
 74     },
 75 
 76     theme: function() {
 77         this.themeChildViews();
 78     },
 79 
 80     /**
 81      * Applies some style-attributes to the grid.
 82      */
 83     style: function() {
 84         if(this.layout) {
 85             var html = 'class="' + this.layout.cssClass + '"';
 86             return html;
 87         }
 88     }
 89 
 90 });
