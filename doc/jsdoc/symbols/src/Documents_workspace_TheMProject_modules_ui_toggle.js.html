 11 /**
 12  * @class
 13  *
 14  * The root of all toggle views.
 15  *
 16  */
 17 M.ToggleView = M.View.extend(
 18 /** @scope M.ToggleView.prototype */ {
 19 
 20     type: 'M.ToggleView',
 21 
 22     isInFirstState: YES,
 23 
 24     toggleOnClick: NO,
 25 
 26     /**
 27      * Renders a ToggleView and its child views.
 28      */
 29     render: function() {
 30         this.html += '<div id="' + this.id + '">';
 31 
 32         this.renderChildViews();
 33 
 34         this.html += '</div>';
 35         
 36         return this.html;
 37     },
 38 
 39     /**
 40      * This method renders the first child view of the toggle view, based on the isInFirstState
 41      * property: YES = first child view, NO = second child view.
 42      */
 43     renderChildViews: function() {
 44         if(this.childViews) {
 45             var childViews = $.trim(this.childViews).split(' ');
 46             var childViewIndex = this.isInFirstState ? 0 : 1;
 47 
 48             if(this[childViews[childViewIndex]]) {
 49                 if(this.toggleOnClick) {
 50                     this[childViews[childViewIndex]].internalTarget = this;
 51                     this[childViews[childViewIndex]].internalAction = 'toggleView';
 52                 }
 53                 this.html += this[childViews[childViewIndex]].render();
 54             } else {
 55                 M.Logger.log('Please make sure that there are two child views defined for the toggle view!', M.WARN);
 56             }
 57         }
 58     },
 59 
 60     /**
 61      * This method toggles the child views.
 62      */
 63     renderUpdateChildViews: function() {
 64         if(this.childViews) {
 65             var childViews = $.trim(this.childViews).split(' ');
 66             var childViewIndex = this.isInFirstState ? 0 : 1;
 67 
 68             if(this[childViews[childViewIndex]]) {
 69                 if(this.toggleOnClick) {
 70                     this[childViews[childViewIndex]].internalTarget = this;
 71                     this[childViews[childViewIndex]].internalAction = 'toggleView';
 72                 }
 73                 this[childViews[childViewIndex]].clearHtml();
 74                 return this[childViews[childViewIndex]].render();
 75             } else {
 76                 M.Logger.log('Please make sure that there are two child views defined for the toggle view!', M.WARN);
 77             }
 78         }
 79     },
 80 
 81     /**
 82      * This method toggles the child views by first emptying the toggle view's content
 83      * and then rendering the next child view by calling renderUpdateChildViews().
 84      */
 85     toggleView: function() {
 86         this.isInFirstState = !this.isInFirstState;
 87         $('#' + this.id).empty();
 88         $('#' + this.id).html(this.renderUpdateChildViews());
 89         this.theme();
 90     },
 91 
 92     theme: function() {
 93         this.themeChildViews();
 94     }
 95 
 96 });
