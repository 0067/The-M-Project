 11 M.DIALOG = {};
 12 M.DIALOG.ALERT = 'Alert';
 13 M.DIALOG.ACTIONSHEET = 'ActionSheet';
 14 
 15 /**
 16  * @class
 17  *
 18  * The root object for DialogViews.
 19  *
 20  */
 21 M.DialogView = M.View.extend(
 22 /** @scope M.DialogView.prototype */ {
 23 
 24     /**
 25      * The type of this object.
 26      *
 27      * @property {String}
 28      */
 29     type: 'M.DialogView',
 30 
 31     /**
 32      * Contains the ids of the button's within a dialog. They are used to later register a click
 33      * event to all of these views.
 34      *
 35      * @property {Object}
 36      */
 37     buttonIds: [],
 38 
 39     /**
 40      * The dialog's callback, splitted in target / action. It is called once the dialog's
 41      * closing transition did finish.
 42      *
 43      * @property {Object}
 44      */
 45     callback: {},
 46 
 47     /**
 48      * This method controls the process of bringing a dialog to the screen.
 49      */
 50     show: function() {
 51 
 52         /* register the onHide event for this dialog */
 53         $('#' + this.id).live('pagehide', this.bindToCaller(this, this.dialogDidHide));
 54 
 55         /* call the dialog's render() */
 56         this.render();
 57 
 58         /* theme it */
 59         this.theme();
 60 
 61         /* register the buttons at the event dispatcher */
 62         for(var i in this.buttonIds) {
 63             M.Application.eventDispatcher.registerEvents(this.buttonIds[i], 'click');
 64         }
 65 
 66         /* finally show the dialog on the screen */
 67         M.Controller.switchToPage(this, this.transition, NO, NO);
 68         
 69     },
 70 
 71     /**
 72      * This method triggers the styling of the dialog and its sub views.
 73      */
 74     theme: function() {
 75         $('#' + this.id).page();
 76     },
 77 
 78     alert: function(obj) {
 79         M.AlertDialogView.design(obj).show();
 80     },
 81 
 82     confirm: function(obj) {
 83         M.ConfirmDialogView.design(obj).show();
 84     },
 85 
 86     actionSheet: function(obj) {
 87          M.ActionSheetDialogView.design(obj).show();
 88     },
 89 
 90     dialogWillClose: function(id) {
 91         var button = M.ViewManager.getViewById(id);
 92         if(this[button.role] && this[button.role].target && this[button.role].action) {
 93             this.callback.target = this[button.role].target;
 94             this.callback.action = this[button.role].action;
 95         } else if (this.buttons && this.buttons[button.role] && this.buttons[button.role].target && this.buttons[button.role].action) {
 96             this.callback.target = this.buttons[button.role].target;
 97             this.callback.action = this.buttons[button.role].action;
 98         }
 99     },
100 
101     dialogDidHide: function() {
102         if(this) {
103             if(this.callback && this.callback.target && this.callback.action) {
104                 this.callback.target[this.callback.action]();
105             }
106             this.destroy();
107         }
108     }
109 
110 });
