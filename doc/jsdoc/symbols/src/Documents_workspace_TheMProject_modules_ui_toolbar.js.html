 11 M.TOP = 'header';
 12 M.BOTTOM = 'footer';
 13 M.LEFT = 0;
 14 M.CENTER = 1;
 15 M.RIGHT = 2;
 16 
 17 /**
 18  * @class
 19  *
 20  * The root object for ToolbarViews.
 21  *
 22  */
 23 M.ToolbarView = M.View.extend(
 24 /** @scope M.ToolbarView.prototype */ {
 25 
 26     /**
 27      * The type of this object.
 28      *
 29      * @type String
 30      */
 31     type: 'M.ToolbarView',
 32 
 33     /**
 34      * Mapping to value attribute.
 35      * text property is mixed in when extended.
 36      */
 37     //value: this.text,
 38 
 39      /**
 40      * Defines the position of the Toolbar.
 41      *
 42      * default: M.TOP => is a header bar
 43      *
 44      * @type String
 45      */
 46     anchorLocation: M.TOP,
 47 
 48     /**
 49      * Display an auto-generated back-button on any new page.
 50      *
 51      * @type Boolean
 52      */
 53     showBackButton: NO,
 54 
 55     /**
 56      * Renders a toolbar as a div tag with corresponding data-role attribute and inner h1 child tag
 57      * (representing the title of the header)
 58      */
 59     render: function() {
 60         this.html += '<div id="' + this.id + '" data-nobackbtn="' + !this.showBackButton + '" data-role="' + this.anchorLocation + '" data-position="fixed">';
 61 
 62         this.renderChildViews();
 63 
 64         this.html += '</div>';
 65 
 66         return this.html;
 67     },
 68 
 69     renderUpdate: function() {
 70         $('#' + this.id + ' h1').html(this.value);
 71     },
 72 
 73     /**
 74      * Triggers render() on all children or simply display the value as a label,
 75      * if it is set.
 76      */
 77     renderChildViews: function() {
 78         if(this.value) {
 79             this.html += '<h1>' + this.value + '</h1>';
 80         } else if (this.childViews) {
 81             var childViews = $.trim(this.childViews).split(' ');
 82 
 83             /* A ToolbarView accepts only 3 childViews, one for each location: left, center, right */
 84             if(childViews.length > 3) {
 85                 M.Logger.log('To many childViews defined for toolbarView.', M.WARN);
 86                 return;
 87             }
 88 
 89             for(var i in childViews) {
 90                 var view = this[childViews[i]];
 91                 switch (view.anchorLocation) {
 92                     case M.LEFT:
 93                         this.html += '<div class="ui-btn-left">';
 94                         this.html += view.render();
 95                         this.html += '</div>';
 96                         break;
 97                     case M.CENTER:
 98                         this.html += '<h1>';
 99                         this.html += view.render();
100                         this.html += '</h1>';
101                         break;
102                     case M.RIGHT:
103                         this.html += '<div class="ui-btn-right">';
104                         this.html += view.render();
105                         this.html += '</div>';
106                         break;
107                 }
108             }
109         }
110     },
111 
112     /**
113      * This method triggers the styling of the toolbar and its subviews.
114      */
115     theme: function() {
116         this.themeChildViews();
117     }
118     
119 });
