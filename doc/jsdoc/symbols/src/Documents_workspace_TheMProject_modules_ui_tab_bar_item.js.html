 11 /**
 12  * @class
 13  *
 14  * The root object for TabBarItemViews.
 15  *
 16  */
 17 M.TabBarItemView = M.View.extend(
 18 /** @scope M.TabBarItemView.prototype */ {
 19 
 20     /**
 21      * The type of this object.
 22      *
 23      * @type String
 24      */
 25     type: 'M.TabBarItemView',
 26 
 27     /**
 28      * Determines whether this TabBarItem is active or not.
 29      *
 30      * @type Boolean
 31      */
 32     isActive: NO,
 33 
 34     internalTarget: null,
 35 
 36     internalAction: 'switchPage',
 37 
 38     render: function() {
 39         this.html += '<li><a id="' + this.id + '"' + this.style() + ' href="#">' + this.value + '</a></li>';
 40 
 41         this.internalTarget = this;
 42         
 43         return this.html;
 44     },
 45 
 46     switchPage: function() {
 47         M.Controller.switchToPage(M.ViewManager.getPage(this.page));
 48     },
 49 
 50     /**
 51      * Applies some style-attributes to the label.
 52      */
 53     style: function() {
 54         var html = '';
 55         if(this.isActive) {
 56             html += ' class="';
 57             html += 'ui-btn-active';
 58             html += '"';
 59         }
 60         if(this.icon) {
 61             html += ' data-icon="';
 62             html += this.icon;
 63             html += '" data-iconpos="top"';
 64         }
 65         return html;
 66     }
 67     
 68 });
