 11 M.STATE_UNDEFINED = 'state_undefined';
 12 M.STATE_NEW = 'state_new';
 13 M.STATE_INSYNCPOS = 'state_insyncpos';
 14 M.STATE_INSYNCNEG = 'state_insyncneg';
 15 M.STATE_LOCALCHANGED = 'state_localchange';
 16 M.STATE_VALID = 'state_valid';
 17 M.STATE_INVALID = 'state_invalid';
 18 M.STATE_DELETED = 'state_deleted';

 23 /**
 24  * @class
 25  *
 26  * The root class for every model.
 27  *
 28  */
 29 M.Model = M.Object.extend(
 30 /** @scope M.Model.prototype */ { 
 31     /**
 32      * The type of this object.
 33      *
 34      * @property {String}
 35      */
 36     type: 'M.Model',
 37 
 38     /**
 39      * The name of the model.
 40      *
 41      * @property {String}
 42      */
 43     name: '',
 44 
 45     /**
 46      * Unique identifier for the model record.
 47      *
 48      * Note: Unique doesn't mean that this id is a global unique ID, it is just unique
 49      * for records of this type of model.
 50      *
 51      * @property {Number}
 52      */
 53     id: null,
 54 
 55     /**
 56      * The model's record defines the properties that are semantically bound to this model:
 57      * e.g. a person's record is in simplest case: firstname, lastname, age.
 58      *
 59      * @property {Object} record
 60      */
 61     record: null,
 62 
 63     /**
 64      * Object containing all meta information for the object's properties
 65      * @property {Object}
 66      */
 67     __meta: {},
 68 
 69     /**
 70      * Manages records of this model
 71      * @property {Object}
 72      */
 73     recordManager: null,
 74 
 75     /**
 76      *
 77      * @param obj
 78      */
 79     records: null,
 80 
 81     /**
 82      * A constant defining the model's state. Important e.g. for syncing storage
 83      * @property {String}
 84      */
 85     state: M.STATE_UNDEFINED,
 86 
 87 
 88     /**
 89      * determines whether model shall be validated before saving to storage or not.
 90      * @property {Boolean}
 91      */
 92     usesValidation: YES,
 93 
 94     /**
 95      * The model's data provider.
 96      *
 97      * Needs a refactoring, also in connection with storageEngine.
 98      *
 99      * @property {Object}
100      */
101     dataProvider: null,
102 
103     /**
104      * Defines the storage engine to use.
105      *
106      * Needs a refactoring, also in connection with dataProvider.
107      *
108      * @property {String}
109      */
110     storageEngine: M.WebStorage,
111 
112     /**
113      * Creates a new record of the model, means an instance of the model based on the blueprint.
114      * You pass the object's specific attributes to it as an object.
115      *
116      * @param {Object} obj The specific attributes as an object, e.g. {firstname: 'peter', lastname ='fox'}
117      */
118     createRecord: function(obj) {
119         var modelRecord = this.extend({
120             id: obj.id ? obj.id : M.Application.modelRegistry.getNextId(this.name),
121             record: obj
122         });
123         modelRecord.state = obj.state ? obj.state : M.STATE_NEW;
124         delete obj.state;
125         delete modelRecord.record.id;
126         return modelRecord;
127     },
128 
129     /** 
130      * Create defines a new model blueprint. It is passed an object with the model's attributes and the model's business logic
131      * and then the type of data provider to use.
132      *
133      * @param {Object} obj An object defining the model's  
134      * @param {Object} dp The data provider to use, e. g. M.LocalStorageProvider
135      */
136     create: function(obj, dp) {
137         var model = M.Model.extend({
138             __meta: {},
139             name: obj.__name__,
140             dataProvider: dp,
141             usesValidation: obj.usesValidation === null || obj.usesValidation === undefined ? this.usesValidation : obj.usesValidation
142         });
143         delete obj.__name__;
144         delete obj.usesValidation;
145 
146         for(var prop in obj) {
147             if(typeof(obj[prop]) === 'function') {
148                 model[prop] = obj[prop];
149             } else if(obj[prop].type === 'M.ModelAttribute') {
150                 model.__meta[prop] = obj[prop];    
151             }
152         }
153 
154         model.recordManager = M.RecordManager.extend({});
155         model.records = model.recordManager.records;
156 
157         M.Application.modelRegistry.register(model.name);
158 
159         /* Re-set the just registered model's id, if there is a value stored */
160         /* Model Registry stores the current id of a model type into localStorage */
161         var id = localStorage.getItem(model.name);
162         if(id) {
163             M.Application.modelRegistry.setId(model.name, parseInt(id));
164         }
165         return model;
166     }, 
167 
168     /**
169      * Returns a M.ModelAttribute object to map an attribute in our model.
170      *
171      * @param type type of the attribute
172      * @param opts options for the attribute, like required flag or
173      */
174     attr: function(type, opts) {
175         return M.ModelAttribute.attr(type, opts); 
176     },
177 
178     /*
179      * get and set methods for encapsulated attribute access
180      */
181 
182     /**
183      * Get attribute attrName from model
184      * @param {String} attrName the name of the attribute which value to be returned
185      * @return {Object|String} value of attribute
186      */
187     get: function(attrName) {
188         return this.record[attrName];
189     },
190 
191     /**
192      * Set attribute attrName of model with value val
193      * @param {String} attrName the name of the attribute which value to be returned
194      * @param {String|Object} val the new value
195      */
196     set: function(attrName, val) {
197         this.record[attrName] = val;
198     },
199 
200     /**
201      * Validates the model, means calling validate for each property.
202      */
203     validate: function() {
204         var isValid = YES;
205         var validationErrorOccured = NO;
206         /* clear validation error buffer before validation */
207         M.Validator.clearErrorBuffer();
208 
209         /*
210         * validationBasis depends on the state of the model: if the model is in state NEW, all properties (__meta includes all)
211         * shall be considered for validation. if model is in another state, the model's record is used. example: the model is loaded from
212         * a database with only two properties included (select name, age FROM...). record now only contains these two properties but __meta
213         * still has all properties listed. models are valid if loaded from database so when saved again only the loaded properties need to get
214         * validated because all others have not been touched. that's why then record is used.
215         * */
216         var validationBasis = this.state === M.STATE_NEW ? this.__meta : this.record;
217 
218         for (var i in validationBasis) {
219             var prop = this.__meta[i];
220             var obj = {
221                 value: this.record[i],
222                 modelId: this.name + '_' + this.id,
223                 property: i
224             };
225             if (!prop.validate(obj)) {
226                 isValid = NO;
227             }
228         }
229         /* set state of model */
230         /*if(!isValid) {
231             this.state = M.STATE_INVALID;
232         } else {
233             this.state = M.STATE_VALID;   
234         }*/
235         return isValid;
236     },
237 
238     /* CRUD Methods below */
239     /**
240      * Calls the corresponding data provider to fetch data based on the passed query.
241      *
242      * @param {Object} obj The param object with query and callbacks.
243      */
244     find: function(obj){
245         /* extends the given obj with self as model property in obj */
246         return this.dataProvider.find( $.extend(obj, {model: this}) );
247     },
248 
249     /**
250      * Returns all models.
251      */
252     findAll: function(){
253         return this.dataProvider.findAll({model: this});
254     },
255 
256     /**
257      * Create or update a record in storage if it is valid (first check this).
258      *
259      * @param {Object} obj The param object with query and callbacks.
260      */
261     save: function(obj) {
262         obj = obj ? obj: {};
263         if(!this.id) {
264             return;
265         }
266         var isValid = YES;
267 
268         if(this.usesValidation) {
269             isValid = this.validate();
270         }
271 
272         if(isValid) {
273             return this.dataProvider.save($.extend(obj, {model: this}));
274         }
275     },
276 
277     /**
278      * Delete a record in storage.
279      */
280     del: function() {
281         if(!this.id) {
282             return;
283         }
284 
285        var isDel = this.dataProvider.del({model: this});
286         if(isDel) {
287             this.state = M.STATE_DELETED;
288         }
289 
290     }
291 
292 });
